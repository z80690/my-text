### 3.2 双模式执行

```python
class DispatcherAgent(BaseAgent):
    SYNC_MODE = "sync"   # 同步模式（默认）
    ASYNC_MODE = "async" # 异步模式（消息总线）

    def __init__(self, config):
        super().__init__(config)
        self._execution_mode = self.SYNC_MODE
        self._init_core_components()

    def _init_core_components(self):
        from .registry import get_registry
        from .agent_protocol import AgentCallInterface, MessageBus
        self._registry = get_registry()
        self._call_interface = AgentCallInterface(self._registry)
        self._message_bus = MessageBus()
        self._message_bus.register_endpoint("dispatcher", self._handle_message)
        self._register_agents_to_bus()

    def _execute_sync(self, agents, task, context):
        return self._call_interface.parallel_call(agents, task, context)

    def _execute_async(self, agents, task, context):
        results = []
        for agent_id in agents:
            msg = AgentMessage(action="execute", source="dispatcher",
                             target=agent_id, payload={"task": task, "context": context})
            response = self._message_bus.send(msg)
            results.append({"agent_id": agent_id, "status": "success", "response": response})
        return results
```

```python
class SwarmAgent(BaseAgent):
    def __init__(self, config):
        super().__init__(config)
        self._swarm_neighbors = []
        self._information_pheromone = {}
        self._swarm_size = 0

    def _default_execute(self, task, context):
        return self._aggregate_results(self._local_process(task, context),
                                       self._swarm_cooperate(task, context))

    def _classify_task(self, task):
        local = super()._classify_task(task)
        votes = [n._classify_task(task) for n in self._swarm_neighbors[:3]]
        if votes:
            c = Counter(votes).most_common(1)[0]
            if c[1] > len(votes) / 2: return c[0]
        return local

    def _generate_response(self, task, context):
        resps = [super()._generate_response(task, context)]
        resps += [n._generate_response(task, context) for n in self._swarm_neighbors[:5]]
        return f"【蜂群综合】{len(resps)}个智能体协作" if len(resps) > 1 else resps[0]
```

## 五、蜂群模式核心机制

### 去中心化决策

```python
class DecentralizedDecisionMaker:
    def make_decision(self, options, context):
        scores = [self._local_evaluate(options, context)]
        scores += self._collect_neighbor_scores(options, context)
        return self._weighted_vote(options, scores)

    def _weighted_vote(self, options, all_scores):
        weights = [0.5] + [0.1] * len(all_scores[1:])
        totals = [0.0] * len(options)
        for s, w in zip(all_scores, weights):
            for i, x in enumerate(s): totals[i] += x * w
        return options[totals.index(max(totals))]
```

```python
class PheromoneSystem:
    def __init__(self, decay=0.1): self._p, self._decay = {}, decay
    def deposit(self, k, v): self._p[k] = self._p.get(k, 0) + v
    def evaporate(self):
        for k in list(self._p.keys()):
            self._p[k] *= (1 - self._decay)
            if self._p[k] < 0.01: del self._p[k]
    def get(self, k): return self._p.get(k, 0)
```

### 自适应行为

```python
class AdaptiveBehavior:
    def __init__(self, agent):
        self._agent = agent
        self._history = []

    def adapt(self, result):
        self._history.append(result.get('status') == 'success')
        r = [x for x in self._history[-10:] if x]
        if r:
            rate = len(r) / len(self._history[-10:])
            if rate < 0.6 and self._agent._swarm_size < 10: self._agent._swarm_size += 1
            elif rate > 0.9 and self._agent._swarm_size > 3: self._agent._swarm_size -= 1
```

class VectorMatcher:
    @staticmethod
    def cosine_similarity(v1, v2):
        dot = sum(a*b for a,b in zip(v1,v2))
        n1 = math.sqrt(sum(a*a for a in v1))
        n2 = math.sqrt(sum(b*b for b in v2))
        return dot/(n1*n2) if n1 and n2 else 0.0

class IntentRecognizer:
    INTENTS = {"code": ["写代码","debug"], "design": ["设计","架构"],
               "review": ["审核","检查"], "research": ["调研","研究"]}
    def recognize(self, text):
        for i, kws in self.INTENTS.items():
            if any(kw in text for kw in kws): return i
        return "general"
