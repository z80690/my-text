#### 5.2 向量匹配机制

```python
class VectorRouter:
    def __init__(self, vector_db):
        self.vector_db = vector_db
    
    async def match_agent(self, task_description: str, top_k: int = 5) -> list:
        """
        根据任务描述匹配最佳智能体
        
        步骤：
        1. 将任务描述转换为向量
        2. 在向量数据库中搜索相似智能体
        3. 返回匹配度最高的前K个智能体
        """
        task_vector = await self._encode(task_description)
        results = await self.vector_db.search(task_vector, top_k)
        return [r['agent_id'] for r in results]
```

class WorkflowEngine:

        self.templates = {}
        self.executors = {}
    
    async def execute(self, workflow_id: str, template_id: str, context: Dict[str, Any]) -> Dict[str, Any]:
        """执行工作流"""
        template = self.templates.get(template_id)
        if not template:
