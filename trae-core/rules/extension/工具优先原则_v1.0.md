### 3.2 参数验证流程

```python
def validate_tool_parameters(tool_name, params):
    """参数验证流程"""
    # 1. 必填检查
    for required_param in tool_schema[tool_name]['required']:
        if required_param not in params:
            return {"valid": False, "error": f"缺少必填参数: {required_param}"}
    
    # 2. 类型检查
    for param_name, param_def in tool_schema[tool_name]['properties'].items():
        if param_name in params:
            expected_type = param_def['type']
            actual_type = type(params[param_name]).__name__
            type_mapping = {'string': 'str', 'number': 'float', 'integer': 'int', 'boolean': 'bool'}
            if actual_type != type_mapping.get(expected_type, expected_type):
                return {"valid": False, "error": f"{param_name} 类型错误，期望 {expected_type}"}
    
    # 3. 格式检查
    for param_name, param_def in tool_schema[tool_name]['properties'].items():
        if param_name in params and 'format' in param_def:
            value = params[param_name]
            format_type = param_def['format']
            if format_type == 'email' and '@' not in value:
                return {"valid": False, "error": f"{param_name} 不是有效的邮箱格式"}
            # 其他格式验证...
    
    # 4. 范围检查
    for param_name, param_def in tool_schema[tool_name]['properties'].items():
        if param_name in params:
            value = params[param_name]
            if 'minimum' in param_def and value < param_def['minimum']:
                return {"valid": False, "error": f"{param_name} 小于最小值 {param_def['minimum']}"}
            if 'maximum' in param_def and value > param_def['maximum']:
                return {"valid": False, "error": f"{param_name} 大于最大值 {param_def['maximum']}"}
            if 'maxLength' in param_def and len(value) > param_def['maxLength']:
                return {"valid": False, "error": f"{param_name} 超过最大长度 {param_def['maxLength']}"}
    
    # 5. 枚举检查
    for param_name, param_def in tool_schema[tool_name]['properties'].items():
        if param_name in params and 'enum' in param_def:
            if params[param_name] not in param_def['enum']:
                return {"valid": False, "error": f"{param_name} 必须是 {param_def['enum']} 之一"}
    
    return {"valid": True}
```

## 4. 工具错误处理规则（L3-R025.4）

### 4.1 错误类型分类

| 错误类型 | 错误码 | 是否可重试 | 处理策略 |
|---------|--------|-----------|---------|
| 网络错误 | NETWORK_ERROR | 是 | 指数退避重试 |
| 速率限制 | RATE_LIMIT | 是 | 等待指定时间后重试 |
| 参数错误 | INVALID_PARAM | 否 | 修正参数后重新调用 |
| 权限错误 | PERMISSION_DENIED | 否 | 提示用户授权 |
| 资源不存在 | NOT_FOUND | 否 | 提示用户资源不存在 |
| 超时错误 | TIMEOUT | 是 | 重试或提示用户 |
| 服务器错误 | INTERNAL_ERROR | 是 | 重试或降级处理 |

### 4.2 错误响应格式

```json
{
  "success": false,
  "error_code": "ERROR_CODE",
  "error_message": "人类可读的错误信息",
  "retryable": true|false,
  "retry_after": 30,
  "suggestion": "用户可以采取的下一步行动"
}
```

### 4.3 重试机制

```python
async def execute_with_retry(tool, params, max_retries=3):
    """带重试机制的工具执行"""
    for attempt in range(max_retries):
        result = await tool.execute(**params)
        
        if result['success']:
            return result
        
        if not result.get('retryable', False):
            return result
        
        # 指数退避
        retry_delay = 2 ** attempt
        
        if 'retry_after' in result:
            retry_delay = max(retry_delay, result['retry_after'])
        
        await asyncio.sleep(retry_delay)
    
    return {
        "success": False,
        "error_code": "MAX_RETRIES_EXCEEDED",
        "error_message": "重试次数已耗尽",
        "retryable": False,
        "suggestion": "请稍后再试或联系支持人员"
    }
```

### 6.2 输入过滤

```python
def sanitize_input(input_str):
    """输入安全过滤"""
    # 移除潜在危险字符
    dangerous_patterns = [';', '`', '$', '|', '<', '>', "'", '"']
    for pattern in dangerous_patterns:
        input_str = input_str.replace(pattern, '')
    
    # 限制长度
    max_length = 10000
    if len(input_str) > max_length:
        input_str = input_str[:max_length]
    
    return input_str
```

### 7.2 工具选择算法

```python
def select_best_tool(request_type, context):
    """根据请求类型选择最佳工具"""
    mapping = REQUEST_TO_TOOL_MAP.get(request_type)
    
    if not mapping:
        return None
    
    # 获取可用工具
    available_tools = [
        tool for tool in registry['tools'] 
        if tool['name'] in mapping['tools'] and tool['enabled']
    ]
    
    if not available_tools:
        return None
    
    # 优先选择成功率高、响应快的工具
    best_tool = min(
        available_tools,
        key=lambda t: (t['success_rate'], -t['avg_response_time_ms'])
    )
    
    return best_tool['name']
```

def pre_execute_checklist(tool_name, params):
    """执行前检查清单"""
    checks = [
        ("是否有合适的工具可以完成此任务？", has_appropriate_tool(tool_name)),
        ("工具参数是否完整且有效？", validate_tool_parameters(tool_name, params)['valid']),
        ("是否理解工具的返回格式？", understands_return_format(tool_name)),
        ("是否处理了可能的错误场景？", has_error_handling(tool_name)),
        ("是否设置了合理的超时时间？", has_timeout_config(tool_name)),
        ("是否考虑了幂等性？", is_idempotent(tool_name))
    ]
    
    all_passed = all(result for _, result in checks)
    
    return {
        "passed": all_passed,
        "checks": checks
    }
