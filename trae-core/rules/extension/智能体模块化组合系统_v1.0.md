### 3.1 模块扫描器实现

```python
import os
import json
from pathlib import Path

class ModuleScanner:
    """模块扫描器 - 自动发现和加载模块"""
    
    def __init__(self, paths=None):
        self.paths = paths or [".trae/rules/", ".trae/skills/", ".trae/algorithms/", ".trae/workflows/", ".trae/agents/"]
        self.registry = {}
    
    def scan(self):
        """扫描所有模块目录"""
        for path in self.paths:
            self._scan_directory(path)
        return self.registry
    
    def _scan_directory(self, dir_path):
        """扫描单个目录"""
        if not os.path.exists(dir_path):
            return
        
        for file_path in Path(dir_path).rglob("*"):
            if file_path.is_file() and self._is_valid_module(file_path):
                module = self._parse_module(file_path)
                if module:
                    self.registry[module['module_id']] = module
    
    def _is_valid_module(self, file_path):
        """检查文件是否为有效模块"""
        valid_extensions = ('.md', '.py', '.json')
        return file_path.suffix in valid_extensions
    
    def _parse_module(self, file_path):
        """解析模块文件"""
        try:
            if file_path.suffix == '.json':
                with open(file_path, 'r', encoding='utf-8') as f:
                    return json.load(f)
            elif file_path.suffix == '.md':
                return self._parse_markdown_module(file_path)
            elif file_path.suffix == '.py':
                return self._parse_python_module(file_path)
        except Exception as e:
            print(f"Failed to parse module {file_path}: {e}")
            return None
    
    def _parse_markdown_module(self, file_path):
        """解析Markdown模块"""
        # 提取元数据部分
        with open(file_path, 'r', encoding='utf-8') as f:
            content = f.read()
        
        # 尝试从内容中提取模块信息
        module_id = self._extract_module_id(file_path)
        return {
            "module_id": module_id,
            "name": self._extract_name(content),
            "version": "1.0.0",
            "type": self._infer_type(file_path),
            "description": self._extract_description(content),
            "file_path": str(file_path),
            "enabled": True
        }
    
    def _extract_module_id(self, file_path):
        """从文件路径提取模块ID"""
        # 根据目录结构推断模块ID
        parts = file_path.parts
        for i, part in enumerate(parts):
            if part.startswith('.trae'):
                # 提取类型前缀
                if 'rules' in parts[i+1:]:
                    prefix = 'L3-R'
                elif 'skills' in parts[i+1:]:
                    prefix = 'L3-S'
                elif 'algorithms' in parts[i+1:]:
                    prefix = 'L3-M'
                elif 'workflows' in parts[i+1:]:
                    prefix = 'L3-W'
                elif 'agents' in parts[i+1:]:
                    prefix = 'L3-C'
                else:
                    prefix = 'L3-R'
                
                # 生成唯一编号
                return f"{prefix}{str(len(self.registry)+1).zfill(3)}"
    
    def _infer_type(self, file_path):
        """推断模块类型"""
        if 'rules' in str(file_path):
            return 'rule'
        elif 'skills' in str(file_path):
            return 'skill'
        elif 'algorithms' in str(file_path):
            return 'algorithm'
        elif 'workflows' in str(file_path):
            return 'workflow'
        elif 'agents' in str(file_path):
            return 'agent'
        return 'unknown'
```

### 4.1 组合配置解析器

```python
import json
from typing import Dict, List, Any

class AgentComposer:
    """智能体组合器 - 根据配置组合模块创建智能体"""
    
    def __init__(self, registry):
        self.registry = registry
        self.composed_agents = {}
    
    def compose(self, config: Dict[str, Any]) -> 'ComposedAgent':
        """根据配置创建组合智能体"""
        agent_id = config['agent_id']
        
        # 验证配置
        self._validate_config(config)
        
        # 解析模块依赖
        modules = self._resolve_modules(config['modules'])
        
        # 构建工作流
        workflow = self._build_workflow(config['workflow'], modules)
        
        # 创建组合智能体
        agent = ComposedAgent(
            agent_id=agent_id,
            name=config['name'],
            description=config.get('description', ''),
            modules=modules,
            workflow=workflow,
            memory_config=config.get('memory', {})
        )
        
        self.composed_agents[agent_id] = agent
        return agent
    
    def _validate_config(self, config: Dict[str, Any]):
        """验证组合配置"""
        required_fields = ['agent_id', 'name', 'modules', 'workflow']
        for field in required_fields:
            if field not in config:
                raise ValueError(f"缺少必填字段: {field}")
        
        # 验证模块配置
        for module_config in config['modules']:
            if 'id' not in module_config:
                raise ValueError("模块配置缺少 id 字段")
    
    def _resolve_modules(self, module_configs: List[Dict[str, Any]]) -> List:
        """解析并加载模块"""
        resolved = []
        
        for config in module_configs:
            module_id = config['id']
            
            if module_id not in self.registry['modules']:
                raise ValueError(f"模块不存在: {module_id}")
            
            module = self.registry['modules'][module_id]
            
            # 检查版本兼容性
            self._check_version_compatibility(module, config.get('version'))
            
            # 解析依赖
            self._resolve_dependencies(module)
            
            resolved.append({
                'module': module,
                'alias': config.get('alias', module_id),
                'config': config.get('config', {})
            })
        
        return resolved
    
    def _check_version_compatibility(self, module, version_constraint=None):
        """检查版本兼容性"""
        if not version_constraint:
            return True
        
        current_version = module['version']
        # 简化的版本检查逻辑
        if version_constraint.startswith('^'):
            major = version_constraint[1:].split('.')[0]
            current_major = current_version.split('.')[0]
            if current_major != major:
                raise ValueError(
                    f"版本不兼容: 期望 ^ {version_constraint[1:]}, 当前 {current_version}"
                )
        elif version_constraint != current_version:
            raise ValueError(
                f"版本不匹配: 期望 {version_constraint}, 当前 {current_version}"
            )
        
        return True
    
    def _resolve_dependencies(self, module):
        """递归解析模块依赖"""
        for dep_id in module.get('dependencies', []):
            if dep_id not in self.registry['modules']:
                raise ValueError(f"依赖模块不存在: {dep_id}")
            
            dep_module = self.registry['modules'][dep_id]
            if not dep_module.get('enabled', True):
                raise ValueError(f"依赖模块已禁用: {dep_id}")
    
    def _build_workflow(self, workflow_config, modules):
        """构建工作流"""
        if isinstance(workflow_config, str):
            # 简单工作流类型
            workflow_type = workflow_config
            steps = self._generate_default_steps(modules)
        else:
            workflow_type = workflow_config['type']
            steps = workflow_config.get('steps', self._generate_default_steps(modules))
        
        return {
            'type': workflow_type,
            'steps': steps
        }
    
    def _generate_default_steps(self, modules):
        """生成默认步骤"""
        steps = []
        for i, mod in enumerate(modules):
            step = {
                'id': f"step_{i+1}",
                'module': mod['alias'],
                'input_from': steps[i-1]['id'] if i > 0 else None,
                'output_to': steps[i+1]['id'] if i < len(modules)-1 else 'result'
            }
            steps.append(step)
        return steps
```

### 4.2 组合智能体类

```python
class ComposedAgent:
    """组合智能体 - 由多个模块组合而成的智能体"""
    
    def __init__(self, agent_id, name, description, modules, workflow, memory_config):
        self.agent_id = agent_id
        self.name = name
        self.description = description
        self.modules = modules
        self.workflow = workflow
        self.memory_config = memory_config
        self.status = 'initialized'
        self.memory = None
        self.module_instances = {}
    
    async def initialize(self):
        """初始化智能体"""
        self.status = 'initializing'
        
        # 初始化内存系统
        self._init_memory()
        
        # 初始化所有模块
        for mod_config in self.modules:
            module = mod_config['module']
            alias = mod_config['alias']
            
            # 创建模块实例
            instance = await self._create_module_instance(module, mod_config['config'])
            self.module_instances[alias] = instance
        
        self.status = 'ready'
    
    def _init_memory(self):
        """初始化内存系统"""
        memory_type = self.memory_config.get('type', 'short_term')
        ttl = self.memory_config.get('ttl', 3600)
        
        if memory_type == 'short_term':
            self.memory = ShortTermMemory(ttl=ttl)
        elif memory_type == 'long_term':
            self.memory = LongTermMemory(ttl=ttl)
        elif memory_type == 'hybrid':
            self.memory = HybridMemory(ttl=ttl)
    
    async def _create_module_instance(self, module, config):
        """创建模块实例"""
        # 根据模块类型创建对应的实例
        module_type = module['type']
        
        if module_type == 'skill':
            return await self._create_skill_instance(module, config)
        elif module_type == 'rule':
            return await self._create_rule_instance(module, config)
        elif module_type == 'algorithm':
            return await self._create_algorithm_instance(module, config)
        elif module_type == 'workflow':
            return await self._create_workflow_instance(module, config)
        elif module_type == 'agent':
            return await self._create_agent_instance(module, config)
        
        raise ValueError(f"不支持的模块类型: {module_type}")
    
    async def execute(self, task, context=None):
        """执行任务"""
        if self.status != 'ready':
            raise RuntimeError("智能体未就绪")
        
        self.status = 'running'
        results = {}
        
        try:
            # 按工作流执行模块
            for step in self.workflow['steps']:
                module_alias = step['module']
                
                if module_alias not in self.module_instances:
                    raise ValueError(f"模块实例不存在: {module_alias}")
                
                instance = self.module_instances[module_alias]
                
                # 获取输入
                input_data = self._get_step_input(step, results, context)
                
                # 执行模块
                result = await instance.execute(input_data)
                results[module_alias] = result
                
                # 检查错误
                if not result.get('success', True):
                    raise Exception(f"模块 {module_alias} 执行失败: {result.get('error_message')}")
            
            # 聚合结果
            final_result = self._aggregate_results(results)
            self.status = 'ready'
            
            return {
                'success': True,
                'data': final_result,
                'agent_id': self.agent_id,
                'module_results': results
            }
        
        except Exception as e:
            self.status = 'error'
            return {
                'success': False,
                'error_message': str(e),
                'agent_id': self.agent_id,
                'module_results': results
            }
    
    def _get_step_input(self, step, previous_results, context):
        """获取步骤输入"""
        input_from = step.get('input_from')
        
        if input_from and input_from in previous_results:
            return previous_results[input_from]
        
        if context:
            return context
        
        return {}
    
    def _aggregate_results(self, results):
        """聚合所有模块结果"""
        aggregated = {}
        
        for alias, result in results.items():
            if result.get('success') and 'data' in result:
                aggregated[alias] = result['data']
        
        return aggregated
    
    async def shutdown(self):
        """关闭智能体"""
        self.status = 'shutdown'
        
        # 清理模块实例
        for instance in self.module_instances.values():
            if hasattr(instance, 'shutdown'):
                await instance.shutdown()
        
        # 清理内存
        if self.memory:
            await self.memory.clear()
```

### 5.2 事件总线实现

```python
import asyncio
from typing import Callable, Dict, List

class EventBus:
    """事件总线 - 模块间通信机制"""
    
    def __init__(self):
        self.subscribers: Dict[str, List[Callable]] = {}
    
    def subscribe(self, event_type: str, handler: Callable):
        """订阅事件"""
        if event_type not in self.subscribers:
            self.subscribers[event_type] = []
        self.subscribers[event_type].append(handler)
    
    def unsubscribe(self, event_type: str, handler: Callable):
        """取消订阅"""
        if event_type in self.subscribers:
            self.subscribers[event_type].remove(handler)
    
    async def publish(self, event_type: str, **kwargs):
        """发布事件"""
        if event_type not in self.subscribers:
            return
        
        event = {
            'type': event_type,
            'data': kwargs,
            'timestamp': asyncio.get_event_loop().time()
        }
        
        # 异步执行所有处理器
        tasks = []
        for handler in self.subscribers[event_type]:
            task = asyncio.create_task(handler(event))
            tasks.append(task)
        
        await asyncio.gather(*tasks, return_exceptions=True)
```

### 6.1 版本约束解析器

```python
import re

class VersionResolver:
    """版本解析器 - 处理版本约束"""
    
    SEMVER_PATTERN = r'^(\d+)\.(\d+)\.(\d+)(?:-([0-9A-Za-z-]+(?:\.[0-9A-Za-z-]+)*))?(?:\+([0-9A-Za-z-]+(?:\.[0-9A-Za-z-]+)*))?$'
    
    @staticmethod
    def parse_version(version_str: str):
        """解析版本字符串"""
        match = re.match(VersionResolver.SEMVER_PATTERN, version_str)
        if not match:
            raise ValueError(f"无效的版本格式: {version_str}")
        
        return {
            'major': int(match.group(1)),
            'minor': int(match.group(2)),
            'patch': int(match.group(3)),
            'prerelease': match.group(4),
            'build': match.group(5)
        }
    
    @staticmethod
    def satisfies_constraint(version: str, constraint: str) -> bool:
        """检查版本是否满足约束"""
        parsed_version = VersionResolver.parse_version(version)
        
        if constraint.startswith('^'):
            return VersionResolver._check_caret_constraint(parsed_version, constraint[1:])
        elif constraint.startswith('~'):
            return VersionResolver._check_tilde_constraint(parsed_version, constraint[1:])
        elif constraint.startswith('>='):
            return VersionResolver._check_gte_constraint(parsed_version, constraint[2:])
        elif constraint.startswith('>'):
            return VersionResolver._check_gt_constraint(parsed_version, constraint[1:])
        elif constraint.startswith('<='):
            return VersionResolver._check_lte_constraint(parsed_version, constraint[2:])
        elif constraint.startswith('<'):
            return VersionResolver._check_lt_constraint(parsed_version, constraint[1:])
        else:
            # 精确匹配
            return version == constraint
    
    @staticmethod
    def _check_caret_constraint(version, constraint):
        """检查 ^x.y.z 约束（兼容相同major版本）"""
        parsed_constraint = VersionResolver.parse_version(constraint)
        return (version['major'] == parsed_constraint['major'] and
                (version['minor'] > parsed_constraint['minor'] or
                 (version['minor'] == parsed_constraint['minor'] and
                  version['patch'] >= parsed_constraint['patch'])))
    
    @staticmethod
    def _check_tilde_constraint(version, constraint):
        """检查 ~x.y.z 约束（兼容相同major.minor版本）"""
        parsed_constraint = VersionResolver.parse_version(constraint)
        return (version['major'] == parsed_constraint['major'] and
                version['minor'] == parsed_constraint['minor'] and
                version['patch'] >= parsed_constraint['patch'])
    
    @staticmethod
    def _check_gte_constraint(version, constraint):
        """检查 >=x.y.z 约束"""
        parsed_constraint = VersionResolver.parse_version(constraint)
        if version['major'] > parsed_constraint['major']:
            return True
        if version['major'] == parsed_constraint['major']:
            if version['minor'] > parsed_constraint['minor']:
                return True
            if version['minor'] == parsed_constraint['minor']:
                return version['patch'] >= parsed_constraint['patch']
        return False
```

### 6.2 依赖图构建器

```python
from collections import defaultdict

class DependencyGraph:
    """依赖图 - 管理模块依赖关系"""
    
    def __init__(self):
        self.graph = defaultdict(list)
        self.reverse_graph = defaultdict(list)
    
    def add_dependency(self, module_id, dep_id):
        """添加依赖关系"""
        self.graph[module_id].append(dep_id)
        self.reverse_graph[dep_id].append(module_id)
    
    def has_cycle(self) -> bool:
        """检测循环依赖"""
        visited = set()
        rec_stack = set()
        
        def dfs(node):
            visited.add(node)
            rec_stack.add(node)
            
            for neighbor in self.graph[node]:
                if neighbor not in visited:
                    if dfs(neighbor):
                        return True
                elif neighbor in rec_stack:
                    return True
            
            rec_stack.remove(node)
            return False
        
        for node in self.graph:
            if node not in visited:
                if dfs(node):
                    return True
        
        return False
    
    def get_dependency_order(self) -> list:
        """获取依赖顺序（拓扑排序）"""
        in_degree = {node: 0 for node in self.graph}
        
        for node in self.graph:
            for neighbor in self.graph[node]:
                in_degree[neighbor] += 1
        
        queue = [node for node in in_degree if in_degree[node] == 0]
        order = []
        
        while queue:
            node = queue.pop(0)
            order.append(node)
            
            for neighbor in self.graph[node]:
                in_degree[neighbor] -= 1
                if in_degree[neighbor] == 0:
                    queue.append(neighbor)
        
        if len(order) != len(self.graph):
            raise ValueError("检测到循环依赖")
        
        return order
```

```python
import re

class ModuleReferenceParser:
    """模块引用解析器 - 解析 @module 语法"""
    
    # 引用模式
    ABSOLUTE_PATTERN = r'@module\(([A-Z0-9-]+)(?:@([^)]+))?\)'
    RELATIVE_PATTERN = r'@module\((\./.*?)\)'
    TAG_PATTERN = r'@module\(tag:([^\)]+)\)'
    TYPE_PATTERN = r'@module\(type:([^\)]+)\)'
    PATTERN_PATTERN = r'@module\(pattern:([^\)]+)\)'
    
    @staticmethod
    def parse(reference_str: str) -> dict:
        """解析模块引用字符串"""
        # 尝试匹配不同模式
        match = re.match(ModuleReferenceParser.ABSOLUTE_PATTERN, reference_str)
        if match:
            return {
                'type': 'absolute',
                'module_id': match.group(1),
                'version': match.group(2) if match.group(2) else 'latest'
            }
        
        match = re.match(ModuleReferenceParser.RELATIVE_PATTERN, reference_str)
        if match:
            return {
                'type': 'relative',
                'path': match.group(1)
            }
        
        match = re.match(ModuleReferenceParser.TAG_PATTERN, reference_str)
        if match:
            return {
                'type': 'tag',
                'tag': match.group(1)
            }
        
        match = re.match(ModuleReferenceParser.TYPE_PATTERN, reference_str)
        if match:
            return {
                'type': 'type',
                'module_type': match.group(1)
            }
        
        match = re.match(ModuleReferenceParser.PATTERN_PATTERN, reference_str)
        if match:
            return {
                'type': 'pattern',
                'pattern': match.group(1)
            }
        
        raise ValueError(f"无效的模块引用格式: {reference_str}")
    
    @staticmethod
    def resolve(reference: dict, registry) -> list:
        """解析引用为实际模块列表"""
        ref_type = reference['type']
        
        if ref_type == 'absolute':
            module_id = reference['module_id']
            if module_id in registry['modules']:
                return [registry['modules'][module_id]]
            return []
        
        elif ref_type == 'relative':
            # 解析相对路径
            path = reference['path']
            # 简化实现：扫描路径
            return ModuleReferenceParser._scan_path(path, registry)
        
        elif ref_type == 'tag':
            tag = reference['tag']
            if tag in registry['index']['by_tag']:
                return [registry['modules'][mid] for mid in registry['index']['by_tag'][tag]]
            return []
        
        elif ref_type == 'type':
            module_type = reference['module_type']
            if module_type in registry['index']['by_type']:
                return [registry['modules'][mid] for mid in registry['index']['by_type'][module_type]]
            return []
        
        elif ref_type == 'pattern':
            pattern = reference['pattern']
            import fnmatch
            matched = []
            for module_id in registry['modules']:
                if fnmatch.fnmatch(module_id, pattern):
                    matched.append(registry['modules'][module_id])
            return matched
        
        return []
    
    @staticmethod
    def _scan_path(path, registry):
        """扫描路径获取模块"""
        # 简化实现
        return [m for m in registry['modules'].values() if path in m.get('file_path', '')]
```

```python
def validate_module_creation(module_def: dict) -> dict:
    """验证模块创建的检查清单"""
    checks = [

        ("类型属于有效类别", _check_module_type(module_def.get('type'))),
        ("描述信息完整", module_def.get('description') is not None and len(module_def['description']) > 0),
        ("依赖关系已声明", 'dependencies' in module_def),
        ("接口定义明确", _check_interface_definition(module_def.get('interfaces'))),
        ("标签不为空", len(module_def.get('tags', [])) > 0),
        ("启用状态已设置", 'enabled' in module_def),

        ("测试覆盖率达标", module_def.get('test_coverage', 0) >= 80),
        ("性能指标已定义", _check_performance_metrics(module_def.get('performance')))
    ]
    
    all_passed = all(result for _, result in checks)
    
    return {
        'passed': all_passed,
        'checks': checks,
        'failed_checks': [desc for desc, passed in checks if not passed]
    }

def _check_module_id_format(module_id: str) -> bool:
    """检查模块ID格式"""
    if not module_id:
        return False
    pattern = r'^L3-[RSMWC]\d{3}$'
    return bool(re.match(pattern, module_id))

def _check_semver_format(version: str) -> bool:
    """检查语义化版本格式"""
    if not version:
        return False
    pattern = r'^\d+\.\d+\.\d+$'
    return bool(re.match(pattern, version))

def _check_module_type(module_type: str) -> bool:
    """检查模块类型"""
    valid_types = ['rule', 'skill', 'algorithm', 'workflow', 'agent']
    return module_type in valid_types

def _check_interface_definition(interfaces: dict) -> bool:
    """检查接口定义"""
    if not interfaces:
        return False
    return 'inputs' in interfaces and 'outputs' in interfaces

def _check_documentation_link(link: str) -> bool:
    """检查文档链接"""
    if not link:
        return True  # 可选
    return link.startswith('http://') or link.startswith('https://') or link.startswith('./')

def _check_performance_metrics(metrics: dict) -> bool:
    """检查性能指标"""
    if not metrics:
        return False
    required_metrics = ['latency_ms', 'throughput', 'success_rate']
    return all(m in metrics for m in required_metrics)
```
