```python
from abc import ABC, abstractmethod
from typing import Dict, Any, Optional, List, AsyncGenerator
from enum import Enum

class AgentStatus(Enum):
    INITIALIZING = "initializing"
    READY = "ready"
    RUNNING = "running"
    ERROR = "error"
    SHUTTING_DOWN = "shutting_down"
    TERMINATED = "terminated"

class AgentType(Enum):
    REACTIVE = "reactive"      # 响应式（按需执行）
    PROACTIVE = "proactive"    # 主动式（定时/事件驱动）
    HYBRID = "hybrid"          # 混合模式

class IAgent(ABC):
    """智能体标准接口"""
    
    @abstractmethod
    def get_id(self) -> str:
        """获取智能体唯一标识"""
        pass
    
    @abstractmethod
    def get_metadata(self) -> Dict[str, Any]:
        """获取智能体元数据"""
        return {
            "agent_id": self.get_id(),
            "name": "",
            "type": AgentType.REACTIVE.value,
            "version": "1.0.0",
            "capabilities": [],
            "triggers": [],
            "resources": {"cpu": 0.1, "memory": 128},
            "qos_level": "normal"
        }
    
    @abstractmethod
    async def execute(self, task: Dict[str, Any], context: Dict[str, Any]) -> Dict[str, Any]:
        """执行任务"""
        pass
    
    @abstractmethod
    async def stream_execute(self, task: Dict[str, Any], context: Dict[str, Any]) -> AsyncGenerator[Dict[str, Any], None]:
        """流式执行任务"""
        pass
    
    @abstractmethod
    async def health_check(self) -> Dict[str, Any]:
        """健康检查"""
        return {"status": AgentStatus.READY.value, "timestamp": time.time()}
    
    @abstractmethod
    async def shutdown(self, timeout: int = 30) -> None:
        """优雅关闭"""
        pass
    
    @abstractmethod
    async def update_config(self, config: Dict[str, Any]) -> None:
        """动态更新配置"""
        pass
```

```python
class PluginManager:
    """智能体插件管理器"""
    
    def __init__(self):
        self.agents: Dict[str, IAgent] = {}
        self.loaders: Dict[str, Type[IAgent]] = {}
        self._lock = asyncio.Lock()
    
    async def register_loader(self, agent_type: str, loader: Type[IAgent]) -> None:
        """注册智能体加载器"""
        async with self._lock:
            self.loaders[agent_type] = loader
    
    async def load_agent(self, agent_id: str, config: Dict[str, Any]) -> IAgent:
        """加载智能体"""
        async with self._lock:
            agent_type = config.get("type", "reactive")
            if agent_type not in self.loaders:
                raise ValueError(f"Unknown agent type: {agent_type}")
            
            loader = self.loaders[agent_type]
            agent = loader(config)
            self.agents[agent_id] = agent
            await agent.initialize()
            return agent
    
    async def unload_agent(self, agent_id: str) -> None:
        """卸载智能体"""
        async with self._lock:
            if agent_id in self.agents:
                await self.agents[agent_id].shutdown()
                del self.agents[agent_id]
    
    async def get_agent(self, agent_id: str) -> Optional[IAgent]:
        """获取智能体实例"""
        return self.agents.get(agent_id)
    
    async def list_agents(self) -> List[Dict[str, Any]]:
        """列出所有智能体"""
```

```python
            return self._storage[key]

        value = await self._read_from_storage(key)
        if value is not None:
            self._storage[key] = value
            return value
        
        return default

        """设置状态"""
        self._storage[key] = value

        await self._replication_queue.put({
            "type": "set",
            "key": key,
            "value": value,
            "ttl": ttl,

        })

        await self._maybe_trigger_snapshot()
```

```python
class QoSManager:
    """服务质量管理器"""
    
    def __init__(self):
        self._agent_metrics: Dict[str, Dict[str, float]] = {}
        self._qos_thresholds = {
            "critical": {"latency_ms": 50, "error_rate": 0.01, "availability": 0.9999},
            "high": {"latency_ms": 100, "error_rate": 0.05, "availability": 0.999},
            "normal": {"latency_ms": 200, "error_rate": 0.1, "availability": 0.99},
            "low": {"latency_ms": 500, "error_rate": 0.2, "availability": 0.95}
        }
    
    async def evaluate(self, agent_id: str, qos_level: str) -> bool:
        """评估智能体是否满足QoS要求"""
        metrics = self._agent_metrics.get(agent_id, {})
        thresholds = self._qos_thresholds.get(qos_level, self._qos_thresholds["normal"])
        
        # 检查延迟
        if metrics.get("latency_ms", 0) > thresholds["latency_ms"]:
            return False
        
        # 检查错误率
        if metrics.get("error_rate", 0) > thresholds["error_rate"]:
            return False
        
        # 检查可用性
        if metrics.get("availability", 1.0) < thresholds["availability"]:
            return False
        
        return True
    
    async def record_metrics(self, agent_id: str, metrics: Dict[str, float]) -> None:
        """记录智能体性能指标"""
        if agent_id not in self._agent_metrics:
            self._agent_metrics[agent_id] = {}
        
        for key, value in metrics.items():
            # 采用滑动窗口平均
            current = self._agent_metrics[agent_id].get(key, 0)
            self._agent_metrics[agent_id][key] = 0.8 * current + 0.2 * value
```

```python
import time
from typing import Dict, Any, Optional
from uuid import uuid4

class TraceContext:
    """追踪上下文"""
    
    def __init__(self, trace_id: Optional[str] = None, span_id: Optional[str] = None):
        self.trace_id = trace_id or str(uuid4())
        self.span_id = span_id or str(uuid4())
        self.parent_span_id = None
        self.start_time = None
        self.end_time = None
        self.metadata: Dict[str, Any] = {}
        self.children: List["TraceContext"] = []
    
    def start(self):
        """开始追踪"""
        self.start_time = time.time()
    
    def end(self):
        """结束追踪"""
        self.end_time = time.time()
    
    def add_metadata(self, key: str, value: Any):
        """添加元数据"""
        self.metadata[key] = value
    
    def create_child(self) -> "TraceContext":
        """创建子追踪上下文"""
        child = TraceContext(trace_id=self.trace_id)
        child.parent_span_id = self.span_id
        self.children.append(child)
        return child
    
    def get_duration(self) -> float:
        """获取持续时间（毫秒）"""
        if self.start_time and self.end_time:
            return (self.end_time - self.start_time) * 1000
        return 0
    
    def to_dict(self) -> Dict[str, Any]:
        """转换为字典"""
        return {
            "trace_id": self.trace_id,
            "span_id": self.span_id,
            "parent_span_id": self.parent_span_id,
            "start_time": self.start_time,
            "end_time": self.end_time,
            "duration_ms": self.get_duration(),
            "metadata": self.metadata,
            "children": [child.to_dict() for child in self.children]
        }
```

### 5.1 沙箱执行环境

```python
import subprocess
import os
import tempfile
from typing import Dict, Any

class SandboxExecutor:
    """沙箱执行器"""
    
    def __init__(self, config: Dict[str, Any]):
        self._cpu_limit = config.get("cpu_limit", 0.5)
        self._memory_limit_mb = config.get("memory_limit_mb", 256)
        self._network_enabled = config.get("network_enabled", False)
        self._allowed_paths = config.get("allowed_paths", [])
    
    async def execute(self, code: str, language: str = "python") -> Dict[str, Any]:
        """在沙箱中执行代码"""
        with tempfile.TemporaryDirectory() as tmpdir:
            # 创建执行脚本
            script_path = os.path.join(tmpdir, f"script.{language}")
            with open(script_path, "w") as f:
                f.write(code)
            
            # 构建执行命令
            cmd = self._build_command(script_path, tmpdir)
            
            # 执行代码
            result = await self._run_command(cmd, tmpdir)
            
            return {
                "stdout": result.get("stdout", ""),
                "stderr": result.get("stderr", ""),
                "return_code": result.get("return_code", 0),
                "execution_time_ms": result.get("execution_time_ms", 0)
            }
    
    def _build_command(self, script_path: str, work_dir: str) -> list:
        """构建受限执行命令"""
        cmd = ["docker", "run", "--rm",
               "--cpus", str(self._cpu_limit),
               "--memory", f"{self._memory_limit_mb}m",
               "-v", f"{work_dir}:/workspace",
               "-w", "/workspace"]
        
        if not self._network_enabled:
            cmd.append("--network", "none")
        
        # 添加路径限制
        for path in self._allowed_paths:
            cmd.extend(["-v", f"{path}:{path}"])
        
        cmd.append("python:3.11")
        cmd.append(script_path)
        
        return cmd
    
    async def _run_command(self, cmd: list, cwd: str) -> Dict[str, Any]:
        """运行命令"""
        start_time = time.time()
        
        try:
            process = await asyncio.create_subprocess_exec(
                *cmd,
                stdout=subprocess.PIPE,
                stderr=subprocess.PIPE,
                cwd=cwd
            )
            
            stdout, stderr = await process.communicate(timeout=60)
            
            return {
                "stdout": stdout.decode("utf-8"),
                "stderr": stderr.decode("utf-8"),
                "return_code": process.returncode,
                "execution_time_ms": (time.time() - start_time) * 1000
            }
        except asyncio.TimeoutError:
            return {
                "stdout": "",
                "stderr": "Execution timed out",
                "return_code": -1,
                "execution_time_ms": 60000
            }
```

```python
        if not isinstance(task, dict):
            errors.append("任务必须是字典格式")
        
        # 2. 必填字段检查
        required_fields = ["task_id", "description"]
        for field in required_fields:
            if field not in task:
                errors.append(f"缺少必填字段: {field}")
        
        # 3. 大小限制
        task_str = json.dumps(task)
        if len(task_str) > InputValidator.MAX_INPUT_SIZE:
            errors.append(f"任务大小超过限制 ({len(task_str)} > {InputValidator.MAX_INPUT_SIZE})")
        
        # 4. 安全检查
        if InputValidator._contains_malicious_content(task):
            errors.append("检测到恶意内容")
        
        # 5. 数据类型验证
        if "priority" in task and task["priority"] not in ["low", "normal", "high", "critical"]:
            errors.append("priority 必须是 low/normal/high/critical")
        
        return {
            "valid": len(errors) == 0,
            "errors": errors
        }
    
    @staticmethod
    def _contains_malicious_content(task: Dict[str, Any]) -> bool:
        """检测恶意内容"""
        task_str = str(task)
        
        for pattern in InputValidator.FORBIDDEN_PATTERNS:
            if re.search(pattern, task_str):
                return True
        
        return False
    
    @staticmethod
    def sanitize_output(output: str) -> str:
        """清理输出内容（数据脱敏）"""
        # 移除敏感信息
        output = re.sub(r"(?i)(api[_-]?key|secret|password|token)\s*[=:]\s*[^,\s]+", r"\1=***", output)
        
        # 移除IP地址
        output = re.sub(r"\b(?:\d{1,3}\.){3}\d{1,3}\b", "[IP]", output)
        
        # 移除邮箱
        output = re.sub(r"[\w\.-]+@[\w\.-]+\.\w+", "[EMAIL]", output)
        
        return output
```

```python
import json
import time
from typing import Dict, Any

class AuditLogger:
    """审计日志记录器"""
    
    def __init__(self, config: Dict[str, Any]):
        self._log_file = config.get("log_file", "audit.log")
        self._log_level = config.get("log_level", "info")
    
    async def log(self, event_type: str, details: Dict[str, Any]) -> None:
        """记录审计日志"""
        log_entry = {
            "timestamp": time.time(),
            "event_type": event_type,
            "details": details,
            "level": self._log_level
        }
        
        # 写入日志文件
        with open(self._log_file, "a") as f:
            f.write(json.dumps(log_entry) + "\n")
    
    async def log_access(self, user_id: str, action: str, resource: str, success: bool) -> None:
        """记录访问日志"""
        await self.log("access", {
            "user_id": user_id,
            "action": action,
            "resource": resource,
            "success": success
        })
    
    async def log_execution(self, task_id: str, agent_id: str, status: str, duration_ms: float) -> None:
        """记录执行日志"""
        await self.log("execution", {
            "task_id": task_id,
            "agent_id": agent_id,
            "status": status,
            "duration_ms": duration_ms
        })
    
    async def log_config_change(self, user_id: str, config_key: str, old_value: Any, new_value: Any) -> None:
        """记录配置变更日志"""
        await self.log("config_change", {
            "user_id": user_id,
            "config_key": config_key,
            "old_value": str(old_value)[:100],
            "new_value": str(new_value)[:100]
        })
```
